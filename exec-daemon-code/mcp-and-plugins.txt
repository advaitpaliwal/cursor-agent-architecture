    static createClient() {
        return new Client({
            name: "Cursor",
            version: "1.0.0",
        }, {
            capabilities: {
                tools: true,
                prompts: true,
                resources: true,
                logging: false,
                elicitation: {},
            },
        });
    }

    // ELICITATION HANDLER - servers can ask questions during tool execution
    setupElicitationHandler() {
        this.client.setRequestHandler(ElicitRequestSchema, async request => {
            if (!this._currentElicitationProvider) {
                return { action: "decline" };
            }
            const response = await this._currentElicitationProvider.elicit({
                message: request.params.message,
                requestedSchema: request.params.requestedSchema,
            });
            return { action: response.action, content: response.content };
        });
    }

    // CONNECT VIA STREAMABLE HTTP (remote MCP servers)
    static async fromStreamableHttp(serverName, config, tokenStorage, headers, authRedirectUrl, fetch) {
        // Sets up OAuth provider with PKCE flow
        // Uses StreamableHTTPClientTransport
        // Handles UnauthorizedError -> sets state to "requires_authentication"
        // with callback to exchange auth code for tokens
        const transport = new StreamableHTTPClientTransport(url, {
            authProvider: oauthProvider,
            requestInit: {
                headers: { "User-Agent": "Cursor/1.0.0", ...(headers ?? {}) },
            },
            fetch,
        });
        const client = mcp_McpSdkClient.createClient();
        await client.connect(transport);
        return new mcp_McpSdkClient(serverName, client, {
            initialState: { kind: "ready" },
            authProvider: oauthProvider,
            config,
        });
    }

    // CONNECT VIA STDIO (local command-based MCP servers)
    static async fromCommand(ctx, serverName, config, env) {
        const transport = new StdioClientTransport({
            command: config.command,
            args: config.args ?? [],
            env,
            cwd: config.cwd,
            stderr: "ignore",
        });
        const client = mcp_McpSdkClient.createClient();
        await client.connect(transport);
        return new mcp_McpSdkClient(serverName, client, {
            initialState: { kind: "ready" },
            config,
        });
    }

    // TOOL EXECUTION - uses lock to serialize calls (elicitation context)
    async callTool(ctx, name, args, toolCallId, elicitationProvider) {
        // Acquires lock to ensure exclusive access to elicitation provider
        // Stores tool call context for elicitation correlation
        this._currentToolCallId = toolCallId;
        this._currentToolName = name;
        this._currentElicitationProvider = elicitationProvider;
        const response = await this.client.callTool({ name, arguments: args });
        return { content: response.content, isError: response.error !== undefined };
    }

    // LIST TOOLS - paginated with cursor
    async getTools(ctx) {
        return this.tools.get(async () => {
            let cursor;
            const tools = [];
            while (true) {
                const response = await this.client.listTools({ cursor });
                tools.push(...response.tools.map(tool => ({
                    ...tool,
                    inputSchema: tool.inputSchema,
                    outputSchema: tool.outputSchema,
                })));
                cursor = response.nextCursor;
                if (cursor === undefined) break;
            }
            return tools;
        });
    }

    // LIST RESOURCES - paginated, checks server capabilities first
    async listResources(ctx) {
        const capabilities = this.client.getServerCapabilities();
        if (!capabilities?.resources) return { resources: [] };
        // ... paginated fetch
    }

    // LIST PROMPTS - checks capabilities
    async listPrompts(ctx) {
        const capabilities = this.client.getServerCapabilities() ?? { prompts: false };
        if (!capabilities.prompts) return [];
        const response = await this.client.listPrompts();
        return response.prompts.map(prompt => ({
            name: prompt.name,
            description: prompt.description,
            arguments: prompt.arguments?.map(arg => ({
                name: arg.name, description: arg.description, required: arg.required,
            })),
        }));
    }
}

class cache_FileBasedMcpCacheStore {

    constructor(mcpCachePath) {
        this.mcpCachePromise = this.loadMcpCache();
    }

    async loadCache(identifier) {
        const mcpCache = await this.mcpCachePromise;
        return mcpCache[identifier] ?? undefined;
    }

    async saveCache(identifier, definition) {
        const mcpCache = await this.mcpCachePromise;
        mcpCache[identifier] = definition;
        await mkdir(path.dirname(this.mcpCachePath), { recursive: true });
        await writeFile(this.mcpCachePath, JSON.stringify(mcpCache, null, 2));
    }
}

class CachedMcpClient {
    // Wraps a clientPromise (still connecting) with cached data
    // Eagerly fetches tools, resources, prompts and saves to cache
    // Returns cached data while client is still connecting

    constructor(ctx, serverName, clientPromise, cacheEntry, cache) {
        // Eagerly fetch tools and save
        this.toolsPromise = this.clientPromise.then(async client => {
            const tools = await client.getTools(ctx);
            await this.cache.saveCache(this.serverName, { ...existing, tools });
            return tools;
        });
        // Similarly for resources and prompts (serialized saves)
    }

    async getTools(ctx) {
        // 1. Check if promise already resolved
        const currentValue = getIfReady(this.toolsPromise);
        if (currentValue !== undefined) return currentValue;
        // 2. Return cached entry if available
        if (this.cacheEntry !== undefined) return this.cacheEntry.tools;
        // 3. Wait for actual fetch
        return await this.toolsPromise;
    }
}

        this.changeListeners.add(listener);
        return () => { this.changeListeners.delete(listener); };
    }

    getClients() { return this.clients; }
    getClient(name) { return this.clients[name]; }
    setClient(name, client) { this.clients[name] = client; this.fireDidChange(); }
    deleteClient(name) { delete this.clients[name]; this.fireDidChange(); }

    // Build unified tool set from all clients
    async getToolSet(ctx) {
        const clientTools = await asyncMapValues(
            Object.entries(this.clients),
            ([key, client]) => client.getTools(ctx)
                .then(tools => tools.map(tool => ({
                    ...tool, clientKey: key, clientName: client.serverName, client
                })))
                .catch(() => []),
            { max: 4 }  // max 4 concurrent
        );
        // Tool naming: `${clientName}-${toolName}`
        const toolsMap = {};

            toolsMap[`${tool.clientName}-${tool.name}`] = {
                definition: { ...tool, providerIdentifier: tool.clientName, toolName: tool.name },

                    const elicitationProvider = elicitationFactory?.createProvider(
                        tool.clientName, tool.name, toolCallId
                    );
                    return tool.client.callTool(ctx, tool.name, args, toolCallId, elicitationProvider);
                },
            };
        }
        return new ExecutableMcpToolSet(toolsMap);
    }

    async getInstructions(ctx) {

--- Server Loading ---
async function loadServer(ctx, serverName, server, tokenStorage, middlewareSettings, authRedirectUrlGenerator) {
    // Middleware chain pattern
    let next = async (ctx, currentServer) => {
        if ("command" in currentServer) {
            return await loadCommandBasedServerWithElicitation(ctx, serverName, currentServer);
        }
        if ("url" in currentServer) {
            return await loadHttpServer(ctx, serverName, currentServer, tokenStorage, authRedirectUrl);
        }
        throw new Error(`Invalid server`);
    };
    for (const middleware of middlewares) {
        const previousNext = next;
        next = async (ctx, server) => middleware.load(ctx, serverName, server, configPath, previousNext);
    }
    return await next(ctx, server);
}

    tokenStorage;
    middlewares;

    async load(ctx, cache = true) {
        const config = await this.configPromise;
        // Load all servers, skip disabled/unapproved
        await Promise.all(Object.entries(config.mcpServers).map(async ([identifier]) => {
            try {
                const client = await this.loadClient(ctx, identifier);
                clients[identifier] = client;
            } catch (error) {
                if (error instanceof McpServerDisabledError) return;
                if (error instanceof McpServerNotApprovedError) return;
                console.error(`Failed to load MCP server "${identifier}":`, error);
            }
        }));
        return new McpManager(clients, this.elicitationFactory);
    }
}

class LocalMcpToolExecutor {
    toolSet;                    // MCP tool set from McpManager
    permissionsService;         // Checks shouldBlockMcp
    pendingDecisionProvider;    // For user approval prompts
    projectDir;
    fileOutputThresholdBytes;   // MCP_TEXT_FILE_THRESHOLD
    elicitationFactory;

    async execute(parentCtx, args) {
        // 1. CHECK PERMISSIONS
        const blockReason = await this.permissionsService.shouldBlockMcp(ctx, args);
        if (blockReason) {
            const blockResult = await handleBlockReason(blockReason, {
                onNeedsApproval: async (approvalReason, approvalDetails) => {
                    // Request user approval via pendingDecisionProvider
                    const result = await this.pendingDecisionProvider.requestApproval({
                        type: OperationType.Mcp,
                        details: { name, toolName, providerIdentifier, args },
                        toolCallId: args.toolCallId,
                    });
                    if (!result.approved) {
                        return McpResult({ case: "rejected", value: McpRejected(...) });
                    }
                    return null; // Continue with execution
                },
                onUserRejected: reason => McpResult({ case: "error", value: McpError(...) }),
                onPermissionDenied: (errorMessage, metadata) =>
                    McpResult({ case: "permissionDenied", value: McpPermissionDenied(...) }),
            });
            if (blockResult) return blockResult;
        }

        // 2. EXECUTE TOOL
        const toolSet = await this.toolSet.getToolSet(ctx);
        result = await toolSet.execute(args.name, parsedArgs, args.toolCallId, this.elicitationFactory);

        // 3. PROCESS RESULTS
        // - Text content: may write to file if > threshold
        // - Image content: resize if needed
        // - Resource content: extract text
        // - Returns McpSuccess with content items
    }
}

async function setupDaemon(options) {
    // options includes: mcpConfig (JSON), httpMcpToolsFile, sandboxPolicyJson, etc.

    // 1. Parse MCP config from CLI option
    let mcpConfig;
    if (mcpConfigJson) {
        mcpConfig = config_mcpConfigSchema.parse(JSON.parse(mcpConfigJson));
    }

    // 2. Load stdio MCP servers
    const customClients = {};
    for (const [serverName, serverConfig] of Object.entries(mcpConfig.mcpServers)) {
        const tokenStorage = createEphemeralScopedTokenStorage();
        customClients[serverName] = await loadServer(ctx, serverName, serverConfig, tokenStorage);
    }

    // 3. Create MCP manager
    const mcpManager = await mcpLoaderWithCustomClients.load(globalContext);
    const stdioMcpLease = new ManagerMcpLease(mcpManager);

    // 4. Load HTTP MCP tools from file (passed by Cursor IDE)
    if (httpMcpToolsJson) {
        httpMcpLease = new StaticMcpLease(JSON.parse(httpMcpToolsJson));
    }

    // 5. Create McpFileSystemWriter if HTTP tools exist
    if (httpMcpLease) {
        const combinedMcpLease = new CombinedMcpLease([stdioMcpLease, httpMcpLease]);
        mcpFileSystemWriter = new McpFileSystemWriter(combinedMcpLease, projectDir, {
            loggerBackend: globalContext.get(loggerKey),
        });
    }

    // 6. Setup permissions
    const permissionsService = new DaemonPermissionsService(defaultSandboxPolicy);

    // 7. Setup hooks
    const hooksConfigPaths = getHooksConfigPaths(projectDir);
    const configLoader = new HooksConfigLoader(fileReader, hooksConfigPaths);
    const hooksConfig = await configLoader.load();
    const hookExecutor = new CliHooksExecutor(
        hooksConfig, projectDir, globalHookContext,
        hooksTerminalExecutor, options.promptHookClient
    );

    // 8. Create resource provider
    const baseResources = new LocalResourceProvider({
        permissionsService,
        mcpLease,
        projectDir,
        getMcpFileSystemOptions: mcpFileSystemWriter
            ? async (ctx, options) => mcpFileSystemWriter.getMcpFileSystemOptions(ctx, options)
            : undefined,
    });

    // 9. Wrap with hooks
    const resources = new ListableHooksResourceAccessor(
        redactedResources, hookExecutor, ...
    );

    return { resources, hookExecutor, mcpFileSystemWriter, ... };
}

1. TWO MCP DISCOVERY MODES:
   a) File System Mode: McpFileSystemWriter writes tool definitions to disk
      under ~/.cursor/projects/<slug>/mcps/<server>/tools/<tool>.json

   b) Meta-Tool Mode: GetMcpTools/CallMcpTool wrapper tools
      Agent calls GetMcpTools to discover, CallMcpTool to execute.
      Enabled via mcp_meta_tool_options in ConversationSettings.

2. MCP SERVER TYPES:
   a) Stdio (command-based): Local process via StdioClientTransport
   b) StreamableHTTP (remote): HTTP+SSE via StreamableHTTPClientTransport with OAuth
   c) HTTP tools from file: Static tool definitions passed by Cursor IDE

   - DaemonPermissionsService: Permissive (cloud exec-daemon)
   - InteractivePermissionsService: Full interactive (IDE)
   - Team admin controls overlay
   - Per-tool allowlist/denylist

   - Claude Code hooks (settings.json format, auto-transformed)
   - Deduplication between the two
   - Multiple config sources: enterprise, team, user, project
