-- Network policy args for sandbox helper --
function buildNetworkPolicyArgs(sandboxPolicy) {
    const args = [];
    if (sandboxPolicy.type === "insecure_none") return args;

    const policy = sandboxPolicy.networkPolicy;
    if (policy === undefined) return args;
    if (!network_policy_utils_isNetworkEnabledByPolicy(policy)) return args;

    // When default is "allow" with no deny list, skip proxy
    // (proxy breaks programs that don't respect HTTP_PROXY)
    const hasDenyList = policy.deny !== undefined && policy.deny.length > 0;
    if (policy.default === "allow" && !hasDenyList) return args;

    // Serialize policy to inline JSON
    const policyJson = serializeNetworkPolicy(policy);
    args.push("--policy-json", policyJson);

    if (sandboxPolicy.networkPolicyStrict === false) {
        args.push("--policy-strict=false");
    }
    return args;
}

function sandbox_spawnInSandbox(command, args = [], options = {}, sandboxPolicy) {
    options.env = filterElectronEnv(options.env);

    // Apply shared build cache env vars if enabled
    if (sandboxPolicy.enableSharedBuildCache) {
        options = {
            ...options,
            env: {
                ...filterElectronEnv(process.env),
                ...getSandboxCacheEnv(),
                ...options.env,
            },
        };
    }

    // Use a lightweight capability check
    if (sandboxPolicy.type !== "insecure_none") {
        if (isSandboxHelperSupported()) {
            return spawnWithSandboxHelper(command, args, options, sandboxPolicy);
        }
        throw new Error(`Sandbox policy '${sandboxPolicy.type}' is not supported...`);
    }

    // Fallback: no sandbox
    return child_process.spawn(command, args, options);
}

class LocalShellExecutor {
    permissionsService;
    coreExecutor;
    ignoreService;

    constructor(permissionsService, coreExecutor, ignoreService, _pendingDecisionProvider) {
        this.permissionsService = permissionsService;
        this.coreExecutor = coreExecutor;
        this.ignoreService = ignoreService;
    }

    async execute(parentCtx, args) {
        const command = args.command;
        const policy = args.requestedSandboxPolicy
            ? convertProtoToInternalPolicy(args.requestedSandboxPolicy)
            : undefined;
        const workingDirectory = args.workingDirectory || (await this.coreExecutor.getCwd());
        const timeout = args.timeout || 30000;
        const resolvedWorkingDir = resolvePath(workingDirectory);

        if (!args.parsingResult) {
            return ShellResult({ case: "spawnError", ... });
        }

        // Unified approval gating (skip if skip_approval is true)
        let effectivePolicy;
        if (!args.skipApproval) {
            const decision = await this.permissionsService.shouldBlockShellCommand(
                ctx, command, {
                    workingDirectory: resolvedWorkingDir,
                    timeout,
                    parsingResult: args.parsingResult,
                    toolCallId: args.toolCallId,
                }, policy
            );
            if (decision.kind === "block") {
                // Check if readonly block vs rejected
                if (decision.reason.isReadonly) {
                    return ShellResult({ case: "permissionDenied", ... });
                }
                return ShellResult({ case: "rejected", ... });
            }
            effectivePolicy = decision.policy;
        } else {
            effectivePolicy = policy;
        }

        // Add cursorignore mapping to effective policy
        if (effectivePolicy && effectivePolicy.type !== "insecure_none") {
            const ignoreMapping = await this.ignoreService.getCursorIgnoreMapping();
            effectivePolicy = { ...effectivePolicy, ignoreMapping };
        }

        return this.executeWithPolicy(ctx, args, workingDirectory, timeout, effectivePolicy);
    }

    async executeWithPolicy(ctx, args, workingDirectory, timeout, policy) {
        const startTime = Date.now();
        const timeoutController = new AbortController();
        const combinedSignal = ctx.signal
            ? AbortSignal.any([ctx.signal, timeoutController.signal])
            : timeoutController.signal;
        const timer = setTimeout(() => timeoutController.abort(), timeout);

        let stdout = "";
        let stderr = "";
        let exitCode = null;
        let aborted = false;
        let outputLocation;

        // Wrap policy as merge sources for shell-exec
        const sandboxPolicyMergeSources = policy ? { perRepo: policy } : undefined;

        for await (const event of this.coreExecutor.execute(ctx, {
            command: args.command,
            workingDirectory,
            signal: combinedSignal,
            toolCallId: args.toolCallId,
            sandboxPolicy: sandboxPolicyMergeSources,
            fileOutputThresholdBytes: args.fileOutputThresholdBytes,
            pipeStdin: false,
        })) {
            if (event.type === "stdout") stdout += event.data;
            else if (event.type === "stderr") stderr += event.data;
            else if (event.type === "exit") {
                exitCode = event.code != null ? event.code | 0 : null;
                outputLocation = event.outputLocation;
                aborted = event.aborted;
            }
        }

        clearTimeout(timer);
        const executionTime = Date.now() - startTime;
        return this.buildResultFromExit({
            command: args.command,
            workingDirectory,
            exitCode: aborted ? -1 : (exitCode ?? -1),
            signal: aborted ? "SIGTERM" : "",
            stdout, stderr, executionTime,
            sandboxPolicy: policy,
            outputLocation,
        });
    }
}

class LocalReadExecutor {
    permissionsService;
    workspacePath;

    async execute(ctx, args) {
        const filePath = args.path;
        const resolvedPath = resolvePath(filePath, this.workspacePath);

        // Permission check with 3s watchdog
        const shouldBlock = await watchdog(ctx, "permissionsService.shouldBlockRead", 3_000,
            () => this.permissionsService.shouldBlockRead(resolvedPath)
        );
        if (shouldBlock) {
            return ReadResult({ case: "permissionDenied", ... });
        }

        // Stat the file
        const stats = await watchdog(ctx, "stat", 3_000, () => fs.stat(resolvedPath));
        if (stats.isDirectory()) return ReadResult({ case: "invalidFile", reason: "Path is a directory" });
        if (!stats.isFile()) return ReadResult({ case: "invalidFile", reason: "Path is neither a file nor a directory" });

        // Determine content type
        const contentInfo = await watchdog(ctx, "getFormatForFile", 3_000,
            () => getFormatForFile(resolvedPath)
        );
        const isImage = contentInfo.isImageFile;
        const isPdf = isPdfFilePath(resolvedPath);
        const isVideo = isVideoFilePath(resolvedPath);

        if (isImage) {
            // Read and resize image if needed
            const imageData = await fs.readFile(resolvedPath);
            const resizedImage = resizeImageBufferIfNeeded(imageData);
            return ReadResult({
                case: "success",
                value: { path: resolvedPath, output: { case: "data", value: resizedImage.data },
                         totalLines: 0, fileSize: BigInt(stats.size), truncated: false },
            });
        }
        else if (isPdf) {
            // Read PDF as binary data
            const pdfData = await fs.readFile(resolvedPath);
            return ReadResult({
                case: "success",
                value: { path: resolvedPath, output: { case: "data", value: pdfData }, ... },
            });
        }
        else if (contentInfo.isBinaryFile && isVideo) {
            // Read video binary for passing to models like Gemini
            const binaryData = await fs.readFile(resolvedPath);
            return ReadResult({ case: "success", value: { output: { case: "data", value: binaryData }, ... } });
        }
        else if (contentInfo.isBinaryFile) {
            return ReadResult({ case: "invalidFile", reason: `Binary files of type ${ext} are not supported` });
        }
        else {
            // TEXT FILE: Read as text
            const fullContent = await watchdog(ctx, "readText", 3_000, () => readText(resolvedPath));
            const totalLines = await watchdog(ctx, "countLines", 3_000, () => countLines(fullContent));

            // Apply offset/limit range if requested
            const rangedRead = tryApplyRequestedReadRange(fullContent, totalLines, args);
            let content = rangedRead.content;
            let truncated = false;

            // Truncate if content exceeds MAX_TEXT_SIZE
            if (content.length > MAX_TEXT_SIZE) {
                content = content.substring(0, MAX_TEXT_SIZE);
                truncated = true;
            }

            return ReadResult({
                case: "success",
                value: {
                    path: resolvedPath,
                    output: { case: "content", value: content },
                    totalLines, fileSize: BigInt(stats.size), truncated,
                    rangeApplied: rangedRead.rangeApplied,
                },
            });
        }
    }
}

-- Range helper --
function tryApplyRequestedReadRange(content, totalLines, args) {
    if (args.offset === undefined && args.limit === undefined) {
        return { content, rangeApplied: false };
    }
    if (content === "") return { content, rangeApplied: false };

    const effectiveOffset = args.offset ?? 1;
    const effectiveLimit = args.limit ?? (effectiveOffset < 0 ? Math.abs(effectiveOffset) : totalLines);
    const startIndex = effectiveOffset < 0
        ? Math.max(0, totalLines + effectiveOffset)
        : Math.max(0, effectiveOffset - 1);

    if (startIndex >= totalLines) return { content, rangeApplied: false };

    const endIndex = Math.min(totalLines, startIndex + effectiveLimit);
    const lines = content.split("\n");
    return { content: lines.slice(startIndex, endIndex).join("\n"), rangeApplied: true };
}

class LocalWriteExecutor {
    fileChangeTracker;
    permissionsService;
    pendingDecisionProvider;
    workspacePaths;
    workspacePath;
    worktreeMainPath;
    onWriteForAiTracking;

    async execute(parentCtx, args) {
        const path = args.path;
        const fileText = args.fileText;
        const fileBytes = args.fileBytes.length > 0 ? args.fileBytes : undefined;
        const resolvedPath = resolvePath(path, this.workspacePath);

        // Skip agent tracking for internal paths
        const shouldSkipAgentTracking = isCursorProjectsInternalPath(resolvedPath, [
            "agent-tools", "swarm-agents"
        ]);

        // Block worktree guard violations
        if (shouldBlockWorktreePath({ targetPath: resolvedPath, ... })) {
            return WriteResult({ case: "error", ... });
        }

        // Read existing content for change tracking
        let beforeContent;
        try { beforeContent = await readText(resolvedPath); } catch { }

        // Permission check
        const shouldBlock = await this.permissionsService.shouldBlockWrite(ctx, resolvedPath, fileText);
        if (shouldBlock) {
            // Handle needs approval, user rejected, permission denied
            const blockResult = await handleBlockReason(shouldBlock, {
                onNeedsApproval: async (approvalReason, approvalDetails) => {
                    const result = await this.pendingDecisionProvider.requestApproval({
                        type: OperationType.Write,
                        details: { path, reason, isNewFile, diffString, before, after, blockReason },
                        toolCallId: args.toolCallId,
                    });
                    if (!result.approved) return WriteResult({ case: "rejected", ... });
                    return null;
                },
                onUserRejected: reason => WriteResult({ case: "rejected", ... }),
                onPermissionDenied: (errorMessage, metadata) => WriteResult({ case: "permissionDenied", ... }),
            });
            if (blockResult !== null) return blockResult;
        }

        // Ensure directory exists
        await fs.mkdir(dir, { recursive: true });

        // Write file content
        let buffer, lines;
        if (fileBytes !== undefined) {
            buffer = fileBytes;       // Binary mode
            lines = 0;
        } else {
            // Text mode: apply line ending normalization
            const textResult = await getBufferForWrite(resolvedPath, fileText, this.workspacePath);
            buffer = textResult.buffer;
            lines = textResult.lines;
        }

        await fs.writeFile(resolvedPath, buffer, { flush: true });
        // Directory sync for durability
        let dirHandle;
        try {
            dirHandle = await fs.open(path.dirname(resolvedPath), O_RDONLY);
            await dirHandle.sync();
        } catch { } finally { await dirHandle?.close(); }

        // Track change
        if (this.fileChangeTracker && fileBytes === undefined && !shouldSkipAgentTracking) {
            await this.fileChangeTracker.trackChange(resolvedPath, beforeContent, fileText, metadata);
        }

        // AI code tracking callback (cursor-blame)
        if (this.onWriteForAiTracking && fileBytes === undefined && !shouldSkipAgentTracking) {
            try {
                await this.onWriteForAiTracking({
                    path: resolvedPath, beforeContent, afterContent: fileText, metadata
                });
            } catch (err) {
                // Best-effort, don't fail the write
            }
        }

        return WriteResult({
            case: "success",
            value: { path: resolvedPath, lines, bytes: buffer.byteLength },
        });
    }
}

    async buildArgs(root, params, mode) {

            }
        } else if (mode === "files_with_matches") {

            args.push("-c", "--with-filename");

            args.push(flag, sortMode);

        args.push("--hidden");  // Include .cursor/rules etc.
        args.push("--regexp", params.pattern);
        args.push("--");
        args.push(params.path ? this.computeRelativeTarget(params.path, root) : ".");
        return args;
    }

    async executeRg(ctx, cwd, args, lineBudget, sandboxPolicy, logContext) {
        const maxBytes = 8 * 1024 * 1024;  // 8MB output cap
        lineBudget = lineBudget ?? LocalGrepExecutor.HARD_MAX_OUTPUT_LINES;

        return new Promise((resolve, reject) => {
            // Spawn ripgrep IN SANDBOX
            let proc = spawnInSandbox(getRipgrepBinaryPath(), args, { cwd }, sandboxPolicy);

            // Timeout
            let timeoutId = setTimeout(() => {
                if (!exited) {
                    reject(new Error(`Timed out after ${LocalGrepExecutor.MAIN_TIMEOUT_MS / 1000}s`));
                    proc.kill();
                }
            }, LocalGrepExecutor.MAIN_TIMEOUT_MS);

            // Stream stdout with line budget enforcement
            proc.stdout?.on("data", (d) => {
                totalStdout += d.length;
                if (lineBudget !== undefined && !lineBudgetExceeded) {
                    // Count '\n' in chunk, find truncation point
                    for (let i = 0; i < d.length; i++) {
                        if (d[i] === 0x0a) {
                            linesInChunk++;
                            if (seenLines + linesInChunk >= lineBudget) {
                                truncateIndex = i + 1;
                                lineBudgetExceeded = true;
                                proc.kill();  // Kill ripgrep when budget exceeded
                                break;
                            }
                        }
                    }
                    seenLines += linesInChunk;
                    const bufferToAdd = truncateIndex >= 0 ? d.subarray(0, truncateIndex) : d;
                    if (totalStdout <= maxBytes) chunksStdout.push(bufferToAdd);
                }
            });

            proc.on("close", (code) => {
                const stdout = Buffer.concat(chunksStdout).toString("utf8");
                const stderr = Buffer.concat(chunksStderr).toString("utf8");
                resolve({ stdout, stderr, exitCode: code ?? 0 });
            });
        });
    }

    async execute(ctx, args, options) {
        const mode = args.outputMode ?? "content";
        const sandboxPolicy = convertProtoToInternalPolicy(args.sandboxPolicy);

        // Run appropriate mode
        switch (mode) {
            case "files_with_matches": return this.runFilesMode(ctx, root, params, sandboxPolicy);
            case "count": return this.runCountMode(ctx, root, params, sandboxPolicy);
            case "content": return this.runContentMode(ctx, root, params, sandboxPolicy);
        }
    }

    // Each mode: buildArgs -> executeRg -> parse output -> filter blocked repos
    async runFilesMode(ctx, root, params, sandboxPolicy) {
        const args = await this.buildArgs(root, params, "files_with_matches");
        const res = await this.executeRg(ctx, root, args, undefined, sandboxPolicy);
        const { files, totalFiles, clientTruncated, ripgrepTruncated } = parseFilesOutput(
            res.stdout, CLIENT_LIMIT_LINES, HARD_MAX_OUTPUT_LINES
        );
        // Filter out files blocked by admin repo rules
        const filteredFiles = [];
        for (const file of files) {
            if (!(await this.ignoreService.isRepoBlocked(file))) {
                filteredFiles.push(file);
            }
        }
        return GrepUnionResult({ case: "files", value: { files: filteredFiles, totalFiles, ... } });
    }
}

function exec(command, args, options) {
    return new Promise((resolve, reject) => {
        child_process.execFile(command, args.map(String), {
            timeout: options?.timeoutMs ?? 30000,
            env: options?.env ? { ...process.env, ...options.env } : undefined,
            encoding: "utf8",
        }, (error, stdout) => {
            if (error) reject(error);
            else resolve(stdout);
        });
    });
}

function execBuffer(command, args, options) {
    return new Promise((resolve, reject) => {
        child_process.execFile(command, args.map(String), {
            timeout: options?.timeoutMs ?? 30000,
            env: options?.env ? { ...process.env, ...options.env } : undefined,
            maxBuffer: 50 * 1024 * 1024,  // 50MB max buffer
            encoding: "buffer",
        }, (error, stdout) => {
            if (error) reject(error);
            else resolve(stdout);
        });
    });
}

const BUTTON_MAP = {
    UNSPECIFIED: "1",  // default left
    LEFT: "1",
    MIDDLE: "2",
    RIGHT: "3",
    BACK: "8",
    FORWARD: "9",

const SCROLL_BUTTON = {
    UNSPECIFIED: 5,  // default down
    UP: 4,
    DOWN: 5,
    LEFT: 6,
    RIGHT: 7,  // (implied from pattern)

class X11Executor {
    config;       // displayNum, resolution, screenshotDelayMs, typingDelayMs, typingBatchSize
    scaler;       // CoordinateScaler instance
    env;          // { DISPLAY: `:${displayNum}` }
    inputEventLogger;  // Optional, for screen recording

    xdotool(args) {
        return exec("xdotool", args.split(" ").filter(Boolean), { env: this.env });
    }

    async execute(actions) {
        const start = Date.now();
        let cursorPosition;
        let lastScreenshot;
        let screenshotTaken = false;

        for (const action of actions) {
            if (action.action.case === "screenshot") {
                await sleep(this.config.screenshotDelayMs);
                lastScreenshot = await this.takeScreenshot();
                screenshotTaken = true;
            }
            else if (action.action.case === "cursorPosition") {
                const result = await this.executeAction(action);
                if (result) cursorPosition = result;
            }
            else {
                const commandStartTime = Date.now();
                await this.executeAction(action);
                const commandEndTime = Date.now();
                // Log for screen recording if active
                if (this.inputEventLogger?.getIsRunning()) {
                    this.inputEventLogger.logEvent(action, commandStartTime, commandEndTime);
                }
            }
        }

        // Auto-screenshot if none were explicitly requested
        if (!screenshotTaken) {
            await sleep(this.config.screenshotDelayMs);
            lastScreenshot = await this.takeScreenshot();
        }

        return {
            success: true, screenshot: lastScreenshot ?? "",
            cursorPosition, actionCount: actions.length,
            durationMs: Date.now() - start,
        };
    }

    async executeAction(action) {
        switch (action.action.case) {
            case "mouseMove": {
                const { x, y } = this.scaleCoordinate(coord);
                await this.xdotool(`mousemove --sync ${x} ${y}`);
                break;
            }
            case "click": {
                const { coordinate, button, count, modifierKeys } = actionOneof.value;
                const buttonNum = BUTTON_MAP[button] ?? "1";
                let clickArgs = buttonNum;
                if (count > 1) clickArgs = `--repeat ${count} --delay 50 ${buttonNum}`;

                const parts = [];
                // Hold modifier keys (split combos like "ctrl+shift")
                const modifiers = modifierKeys?.split("+").filter(Boolean) ?? [];
                for (const mod of modifiers) parts.push(`keydown ${mod}`);

                if (coordinate) {
                    const { x, y } = this.scaleCoordinate(coordinate);
                    parts.push(`mousemove --sync ${x} ${y}`);
                }
                parts.push(`click ${clickArgs}`);

                // Release modifiers in reverse order
                for (const mod of [...modifiers].reverse()) parts.push(`keyup ${mod}`);
                await this.xdotool(parts.join(" "));
                break;
            }
            case "mouseDown": {
                await this.xdotool(`mousedown ${buttonNum}`);
                break;
            }
            case "mouseUp": {
                await this.xdotool(`mouseup ${buttonNum}`);
                break;
            }
            case "drag": {
                const { path, button } = actionOneof.value;
                const scaledPath = path.map(coord => this.scaleCoordinate(coord));
                const commands = [];
                commands.push(`mousemove --sync ${scaledPath[0].x} ${scaledPath[0].y}`);
                commands.push(`mousedown ${buttonNum}`);
                for (let i = 1; i < scaledPath.length; i++) {
                    commands.push(`mousemove --sync ${scaledPath[i].x} ${scaledPath[i].y}`);
                }
                commands.push(`mouseup ${buttonNum}`);
                await this.xdotool(commands.join(" "));
                break;
            }
            case "scroll": {
                const { coordinate, direction, amount, modifierKeys } = actionOneof.value;
                const scrollButton = SCROLL_BUTTON[direction] ?? 5;
                const parts = [];
                const modifiers = modifierKeys?.split("+").filter(Boolean) ?? [];
                for (const mod of modifiers) parts.push(`keydown ${mod}`);
                if (coordinate) {
                    const { x, y } = this.scaleCoordinate(coordinate);
                    parts.push(`mousemove --sync ${x} ${y}`);
                }
                parts.push(`click --repeat ${amount} ${scrollButton}`);
                for (const mod of [...modifiers].reverse()) parts.push(`keyup ${mod}`);
                await this.xdotool(parts.join(" "));
                break;
            }
            case "type":
                await this.typeText(text);
                break;
            case "key": {
                if (holdDurationMs && holdDurationMs > 0) {
                    await this.xdotool(`keydown ${key}`);
                    await sleep(holdDurationMs);
                    await this.xdotool(`keyup ${key}`);
                } else {
                    await this.xdotool(`key -- ${key}`);
                }
                break;
            }
            case "wait":
                await sleep(actionOneof.value.durationMs);
                break;
            case "cursorPosition": {
                const output = await this.xdotool("getmouselocation --shell");
                const xMatch = output.match(/X=(\d+)/);
                const yMatch = output.match(/Y=(\d+)/);
                const displayX = parseInt(xMatch[1], 10);
                const displayY = parseInt(yMatch[1], 10);
                return this.scaler.displayToApi(displayX, displayY);
            }
        }
    }

    async typeText(text) {
        const { typingDelayMs, typingBatchSize } = this.config;
        const segments = text.split(/\r\n|\r|\n/);
        for (let segIdx = 0; segIdx < segments.length; segIdx++) {
            const segment = segments[segIdx];
            // Type in batches of typingBatchSize chars
            for (let i = 0; i < segment.length; i += typingBatchSize) {
                const chunk = segment.slice(i, i + typingBatchSize);
                await exec("xdotool", ["type", "--delay", String(typingDelayMs), "--", chunk],
                    { env: this.env });
            }
            // Press Return for newlines (xdotool type sends Linefeed for \n)
            if (segIdx < segments.length - 1) {
                await this.xdotool("key Return");
            }
        }
    }

    async takeScreenshot() {
        const { width, height } = this.config.resolution.display;
        const buffer = await execBuffer("ffmpeg", [
            "-f", "x11grab",
            "-video_size", `${width}x${height}`,
            "-i", `:${this.config.displayNum}`,
            "-frames:v", "1",
            "-vf", `scale=${this.scaler.apiWidth}:${this.scaler.apiHeight}`,
            "-c:v", "libwebp",
            "-preset", "text",     // Optimized for UI screenshots
            "-lossless", "1",
            "-f", "webp",
            "pipe:1",
        ], { env: this.env });

        // Fix WebP headers: ffmpeg can't seek in pipes, so RIFF/VP8 chunk sizes are 0
        const isSimpleVP8WebP = buffer.length > 20 &&
            buffer.subarray(0, 4).toString() === "RIFF" &&
            buffer.subarray(8, 12).toString() === "WEBP" &&
            buffer.subarray(12, 16).toString() === "VP8 ";
        if (isSimpleVP8WebP) {
            buffer.writeUInt32LE(buffer.length - 8, 4);    // RIFF size
            buffer.writeUInt32LE(buffer.length - 20, 16);  // VP8 chunk size
        }

        return buffer.toString("base64");
    }
}

class CoordinateScaler {
    config;
    xScaleUp; yScaleUp;     // API -> display
    xScaleDown; yScaleDown;  // display -> API

    constructor(config) {
        // Validate aspect ratio match
        const displayRatio = config.display.width / config.display.height;
        const apiRatio = config.api.width / config.api.height;
        if (Math.abs(displayRatio - apiRatio) > 0.02) {
            throw new Error(`Aspect ratio mismatch`);
        }

        this.xScaleUp = config.display.width / config.api.width;     // e.g. 1920/1280 = 1.5
        this.yScaleUp = config.display.height / config.api.height;    // e.g. 1200/800 = 1.5
        this.xScaleDown = config.api.width / config.display.width;    // e.g. 1280/1920 = 0.667
        this.yScaleDown = config.api.height / config.display.height;  // e.g. 800/1200 = 0.667
    }

    apiToDisplay(x, y) {
        return { x: Math.round(x * this.xScaleUp), y: Math.round(y * this.yScaleUp) };
    }

    displayToApi(x, y) {
        return { x: Math.round(x * this.xScaleDown), y: Math.round(y * this.yScaleDown) };
    }
}

class FileChangeTracker {
    workspacePath;
    changes = new Map();   // path -> { path, before, after, metadata }
    listeners = new Set();

    async trackChange(path, before, after, metadata) {
        const existingChange = this.changes.get(path);
        if (existingChange) {
            // Merge: keep original "before" state
            const originalBefore = existingChange.before;
            if (originalBefore === after || (originalBefore === undefined && after === undefined)) {
                // No net change - remove from tracking
                this.changes.delete(path);
            } else {
                this.changes.set(path, { path, before: originalBefore, after, metadata });
            }
        } else {
            this.changes.set(path, { path, before, after, metadata });
        }
        this.notify(change);
    }

    async reject(path) {
        const change = this.changes.get(path);
        if (!change) return;
        if (change.before === undefined && change.after !== undefined) {
            await fs.unlink(path);           // File was created -> delete
        } else if (change.before !== undefined) {
            await writeText(path, change.before);  // Restore original
        }
        this.changes.delete(path);
    }
}

TOOL EXECUTION FLOW:
  Backend (Cursor server)
    -> protobuf RPC over exec stream (ExecutorResource pattern)
    -> exec-daemon (this file)
    -> Local*Executor class
    -> Permission check (permissionsService.shouldBlock*)
    -> Sandbox policy construction (merge hardcoded + admin + per-repo)
    -> spawnInSandbox() or direct fs operations

SANDBOX WRAPPING:
  Shell commands:
    LocalShellExecutor -> coreExecutor.execute() -> spawnInSandbox()
    -> cursorsandbox --sandbox-policy-cwd <cwd> --sandbox-policy <json> -- <shell> -c <cmd>

  Grep:
    LocalGrepExecutor -> executeRg() -> spawnInSandbox(ripgrepBinaryPath, args, ...)
    -> cursorsandbox ... -- <ripgrep-fork> --cursor-ignore ... --regexp <pattern>

  File operations (read/write/delete):
    Direct Node.js fs operations (NOT sandboxed via cursorsandbox)
    Permission checks via permissionsService
    Protected path checking via isProtectedPath() (gitignore-style)

  Computer use:
    X11Executor -> xdotool (mouse/keyboard) + ffmpeg (screenshots)
    NOT sandboxed (runs directly on X11 display)
    Coordinate scaling: API resolution (1280x800) <-> display resolution (1920x1200)

  Fetch:
    Direct Node.js fetch() - network sandboxing at sandbox proxy level

KEY PROPRIETARY ELEMENTS:
  1. cursorsandbox binary - Rust binary implementing Landlock (Linux), seatbelt (macOS)
  2. Forked ripgrep with --cursor-ignore flag (unforgeable ignore rules)
  3. Sandbox policy JSON schema (workspace_readonly / workspace_readwrite / insecure_none)
  4. Network policy proxy (allow/deny lists enforced by sandbox binary)
  5. Protected write patterns preventing agent from modifying its own control plane
  6. File change tracking with undo/redo support (cursor-blame integration)
  7. Polished renderer for screen recording post-processing
