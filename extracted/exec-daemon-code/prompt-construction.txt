1. ARCHITECTURE OVERVIEW
================================================================================

The exec-daemon does NOT contain the actual system prompt text/template. It is
a LOCAL process that:
  - Collects environment data, cursor rules, MCP tools, git info, skills
  - Packages everything into a protobuf `RequestContext` message
  - Sends this via gRPC to the Cursor AI SERVER (aiserver), which assembles
    the actual system prompt text

The prompt construction is split:
  CLIENT (exec-daemon)  ->  builds RequestContext proto  ->  SERVER (aiserver)
  - OS/shell/paths         - rules[]                       - Assembles system prompt
  - Git repos/branches     - tools[] (MCP)                 - Injects <rules> XML
  - Cursor rules (.mdc)    - env (OS, paths, etc)          - Injects <user_info>
  - AGENTS.md / CLOUD.md   - cloudRule                     - Adds tool definitions
  - Skills (SKILL.md)      - agentSkills[]                 - Creates <system_reminder>
  - MCP tool definitions   - customSubagents[]

            gitRepoInfoComplete,
            repositoryInfo: [...],               // Codebase indexing info
            tools: mcp.servers                   // MCP tools flattened
                .flatMap(server => server.tools)
                .map(tool => new McpToolDefinition({
                    name: tool.name,
                    providerIdentifier: tool.providerIdentifier,
                    toolName: tool.toolName,
                    description: tool.description,
                    inputSchema: tool.inputSchema
                        ? Value.fromJson(tool.inputSchema) : undefined,
                })),
            conversationNotesListing,
            sharedNotesListing,
            mcpInstructions: mcp.servers.flatMap(server => server.instructions),
            customSubagents: globalData.customSubagents,
            mcpFileSystemOptions,
            mcpMetaToolOptions,                  // When enabled, uses GetMcpTools/CallMcpTool
            supportsMcpAuth: true,
            commitAttributionMessage: attributeCommitsToAgent ? "enabled" : undefined,
            prAttributionMessage: attributePRsToAgent ? "enabled" : undefined,
            agentSkills: this.getAgentSkills ? await this.getAgentSkills() : [],
        });

        // Cloud rule injected separately (optional)

    const shell = getSuggestedShell(userTerminalHint);

    const projectFolder = projectDir;
    const terminalsFolder = projectDir ? `${projectDir}/terminals` : undefined;
    const agentSharedNotesFolder = projectDir ? `${projectDir}/agent-notes/shared` : undefined;

        osVersion, workspacePaths, shell, sandboxEnabled,
        projectFolder, terminalsFolder, agentSharedNotesFolder,
        agentConversationNotesFolder, agentTranscriptsFolder, timeZone,
        artifactsFolder, secretRedactionEnabled,
        sandboxSupported, sandboxNetworkHasDefaults, sandboxNetworkExplicitAllowlist,

class MergedCursorRulesService {
    cursorRulesServices;  // Array of rule service implementations

    async getAllCursorRules(ctx) {
        // Call all services in parallel
        const allRulesArrays = await Promise.all(
            this.cursorRulesServices.map(s => s.getAllCursorRules(ctx))
        );
        const allRules = allRulesArrays.flat();

        // Dedupe by key - first occurrence wins (Cursor sources processed first)
        // dedupeKey() uses (config root)::(path under config) for .cursor/.claude
        // so same skill/rule in both directories collides and Cursor wins
        const seenKeys = new Set();
        const dedupedRules = [];
        for (const rule of allRules) {
            const key = dedupeKey(rule.fullPath);
            if (!seenKeys.has(key)) {
                seenKeys.add(key);
                dedupedRules.push(rule);
            }
        }
        return dedupedRules;
    }
}

class LocalCursorRulesService {
    // File patterns watched:
    //   ".cursor/rules/**/*.mdc", "AGENTS.md", "CLAUDE.md", "CLAUDE.local.md", ".cursorrules"

    async load(ctx) {
        const cwd = path.resolve(this.rootDirectory);
        const gitRoot = await findGitRoot(ctx, this.gitExecutor, cwd);
        const workspaceRoot = gitRoot || cwd;

        // Load from multiple sources in parallel:
        const allRules = await Promise.all([
            this.loadCursorRulesRule(ctx, workspaceRoot),      // .cursorrules (legacy)
            this.loadRulesFromHierarchy(ctx, cwd),             // Walk up directory tree
            this.loadRulesFromSubdirectories(ctx, cwd),        // Nested .cursor/rules/**/*.mdc
            this.loadMarkdownFilesFromSubdirectories(ctx, cwd) // Nested AGENTS.md/CLAUDE.md
        ]);

        // Dedupe by path, filter excluded, convert to CursorRule proto
        return [...map.values()]
            .filter(rule => !this.shouldExcludeFromCursorRules(rule))
            .map(pr => this.toCursorRule(pr));
    }

    // Walk UP the directory tree loading rules at each level
    async loadRulesFromHierarchy(ctx, startPath) {
        let currentPath = startPath;
        while (true) {
            // Check for .cursor/rules directory
            const rulesDir = path.join(currentPath, ".cursor", "rules");
            if (await this.dirIsDirectory(ctx, rulesDir))
                rules.push(...(await this.loadRulesFromDirectory(ctx, rulesDir)));

            // Optionally check CLAUDE.md and CLAUDE.local.md (third-party extensibility)
            if (this.getThirdPartyExtensibilityEnabled()) {
                rules.push(...(await this.loadRulesFromMarkdownFile(ctx, currentPath, "CLAUDE.md")));
                rules.push(...(await this.loadRulesFromMarkdownFile(ctx, currentPath, "CLAUDE.local.md")));
            }

            // Always check AGENTS.md
            rules.push(...(await this.loadRulesFromMarkdownFile(ctx, currentPath, "AGENTS.md")));

            const parentPath = path.dirname(currentPath);
            if (parentPath === currentPath) break;
            currentPath = parentPath;
        }
        return rules;
    }

    // Rule type classification:
    toCursorRule(rule) {
        if (rule.frontmatter.alwaysApply === true) {
            type = { case: "global", value: new CursorRuleTypeGlobal() };
        } else {
            const globs = parseGlobs(rule.frontmatter.globs);
            if (globs && globs.length > 0) {
                type = { case: "fileGlobbed", value: new CursorRuleTypeFileGlobs({ globs }) };
            } else if (typeof rule.frontmatter.description === "string") {
                type = { case: "agentFetched", value: new CursorRuleTypeAgentFetched({ description }) };
            } else {
                type = { case: "manuallyAttached", value: new CursorRuleTypeManuallyAttached() };
            }
        }
        return new CursorRule({
            fullPath: rule.path,
            content: rule.body,
            type: new CursorRuleType({ type }),
            environments,
            disabledEnvironments,
        });
    }

const CLOUD_SPECIFIC_INSTRUCTIONS_HEADER = "## Cursor Cloud specific instructions";
const CLOUD_SPECIFIC_INSTRUCTIONS_SEPARATOR = `\n\n${CLOUD_SPECIFIC_INSTRUCTIONS_HEADER}\n`;

class LocalCloudRulesService {
    // Watches: [".cursor/CLOUD.md", "AGENTS.md"]

    async load(ctx) {
        const workspaceRoot = path.resolve(this.rootDirectory);
        const cloudRulesPath = path.join(workspaceRoot, ".cursor", "CLOUD.md");
        const agentsRulesPath = path.join(workspaceRoot, "AGENTS.md");

        const [cloudRulesContent, agentsRulesContent] = await Promise.all([
            readRulesFile(cloudRulesPath, "cloudRulesFilename", "CLOUD.md"),
            readRulesFile(agentsRulesPath, "agentsRulesFilename", "AGENTS.md"),
        ]);

        if (!cloudRulesContent && !agentsRulesContent) return null;

        // CLOUD.md only -> wrapped with header
        if (!agentsRulesContent) {
            return `${CLOUD_SPECIFIC_INSTRUCTIONS_HEADER}\n${cloudRulesContent}`;
        }

        // AGENTS.md only -> returned as-is
        if (!cloudRulesContent) {
            return agentsRulesContent;
        }

        // Both -> AGENTS.md content + separator + CLOUD.md content
        return `${agentsRulesContent}${CLOUD_SPECIFIC_INSTRUCTIONS_SEPARATOR}${cloudRulesContent}`;
    }
}

// Message hydration: Reconstruct full conversation from blob store
async function hydrateMessages(ctx, blobStore, state) {
    // 1. Hydrate archived messages from summary_archives (if present)
    const summaryArchives = await Promise.all(
        state.summaryArchives.map(async summaryArchiveRef => {
            const archiveBlob = await blobStore.getBlob(ctx, summaryArchiveRef);
            const archive = ConversationSummaryArchive.fromBinary(archiveBlob);
            return await hydrateBlobIds(ctx, blobStore, archive.summarizedMessages);
        })
    );
    allMessages.push(...summaryArchives.flat());

    // 2. Hydrate current prompt messages
    const promptMessages = await hydrateBlobIds(ctx, blobStore, state.rootPromptMessagesJson);

    // 3. Filter out system messages and summary messages from prompt tail
    const tailMessages = promptMessages.filter(
        msg => msg.role !== "system" && !isSummaryMessage(msg)
    );

    // 4. Concatenate archived + filtered tail
    allMessages.push(...tailMessages);
    return allMessages;
}

// These XML tags are what the SERVER wraps around injected content in the
// system prompt. The exec-daemon strips them when generating transcripts.

function stripXmlTags(text) {
    for (const tag of XML_TAGS_TO_STRIP) {
        const pattern = new RegExp(`<${tag}>[\\s\\S]*?</${tag}>`, "g");
        result = result.replace(pattern, "");
    }
    return result.replace(/\n{3,}/g, "\n\n").trim();
}

// Also strips hidden thinking tags:
function stripHiddenThinkingTags(text) {
    return text
        .replace(/<think>[\s\S]*?<\/think>/gi, "")
        .replace(/<thinking>[\s\S]*?<\/thinking>/gi, "")
        .replace(/\n{3,}/g, "\n\n")
        .trim();
}

12. KEY ARCHITECTURAL INSIGHT
================================================================================

The actual system prompt TEXT is assembled SERVER-SIDE, not in the exec-daemon.
The exec-daemon's role is to:

1. COLLECT all context data (env, rules, tools, git, skills, notes, etc.)
2. PACKAGE it into a RequestContext protobuf message
3. SEND it alongside the ConversationStateStructure to the AI server

The server then:
- Builds the system prompt text using the RequestContext fields
- Wraps rules in <rules> XML tags
- Wraps user info in <user_info> tags
- Wraps open files in <open_and_recently_viewed_files> tags
- Injects system reminders in <system_reminder> tags
- Adds tool definitions (builtin + MCP)
- Applies cursor rules based on type (global=always, fileGlobbed=on match,
  agentFetched=on-demand, manuallyAttached=user-attached)
- Injects cloud rules (AGENTS.md + CLOUD.md) via the cloudRule field

This split architecture means the exec-daemon is a data-collection layer,
while the actual prompt engineering happens on Cursor's servers.

The conversation state uses a blob-store pattern where messages are stored
as binary blobs referenced by ID, allowing efficient persistence and
resumption of conversations.
